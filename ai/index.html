对话管理 self.contexts: Dict[str, DialogueContext] = {} self.current_conversation_id: Optional[str] = None # 系统提示词 self.system_prompt = """你是一个专业的AI对话助手，擅长： 1. 理解用户的真实意图 2. 从知识库中检索相关信息 3. 提供准确、清晰的回答 4. 处理歧义和模糊提问 在回答时： - 优先使用知识库中的信息 - 如果知识库没有相关内容，进行合理推理 - 保持回答的准确性和专业性 """ print("✓ 对话机器人已初始化") print(" - 语义理解模块已加载") print(" - 多知识库系统已启动") print(" - 答案格式化模块已加载") print(" - 敏感词过滤已启用") print(" - 兜底管理已启动") print(f" - LLM状态: {'已连接' if self.use_llm and self.ai_client.chat_client else '未连接'}") # ==================== 对话管理 ==================== def start_conversation(self) -> str: """开始新对话""" conv_id = f"conv_{len(self.contexts)}_{datetime.now().timestamp()}" self.contexts[conv_id] = DialogueContext(conversation_id=conv_id) self.current_conversation_id = conv_id return conv_id def add_user_message(self, content: str, media_path: Optional[str] = None, media_type: Optional[str] = None) -> Message: """添加用户消息""" if not self.current_conversation_id: self.start_conversation() # 语义处理 intent = self.intent_recognizer.recognize(content) corrected_content = self.semantic_corrector.correct(content) msg = Message( id=f"msg_{len(self.contexts[self.current_conversation_id].messages)}", role="user", content=content, timestamp=datetime.now().isoformat(), message_type=media_type or "text", metadata={"media_path": media_path} if media_path else {}, intent=intent.value, corrected_content=corrected_content ) self.contexts[self.current_conversation_id].add_message(msg) return msg def add_assistant_message(self, content: str) -> Message: """添加助手消息""" if not self.current_conversation_id: self.start_conversation() msg = Message( id=f"msg_{len(self.contexts[self.current_conversation_id].messages)}", role="assistant", content=content, timestamp=datetime.now().isoformat(), message_type="text" ) self.contexts[self.current_conversation_id].add_message(msg) return msg # ==================== 增强的搜索功能 ==================== def search_knowledge(self, query: str, kb_names: List[str] = None, doc_type: DocumentType = None, threshold: float = 0.3) -> Tuple[List[SearchResult], str]: """增强的知识库搜索""" # 1. 语义纠错 corrected_query = self.semantic_corrector.correct(query) # 2. 问题重写 rewritten_query = self.question_rewriter.rewrite(corrected_query) # 3. 多知识库检索 results = self.kb_manager.search_across_kbs( rewritten_query, kb_names=kb_names, threshold=threshold, doc_type=doc_type ) # 构建搜索信息 search_info = f"[搜索分析] 原始查询: {query}\n" if corrected_query != query: search_info += f"纠正后: {corrected_query}\n" if rewritten_query != corrected_query: search_info += f"重写后: {rewritten_query}\n" search_info += f"找到 {len(results)} 个相关结果" return results, search_info def get_next_response(self, user_input: str) -> Tuple[str, Dict]: """ 获取下一个响应 - 增强版本，包含： 1. 敏感词过滤 2. 答案拼接和格式化 3. 引用标注 4. 兜底管理 """ if not self.current_conversation_id: self.start_conversation() # ==================== 第一步：敏感词检查 ==================== has_sensitive, found_words = self.sensitive_filter.contains_sensitive_words(user_input) if has_sensitive: fallback_response, fallback_type = self.fallback_manager.get_best_fallback_for_search( results=[], has_sensitive=True ) # 添加用户消息 self.add_user_message(user_input) # 添加回复消息 self.add_assistant_message(fallback_response) return fallback_response, { "user_message": user_input, "assistant_message": fallback_response, "has_sensitive": True, "found_sensitive_words": found_words, "is_fallback": True, "fallback_type": "sensitive", "sources": [], "related_questions": [], "markdown": fallback_response } # ==================== 第二步：添加用户消息 ==================== user_msg = self.add_user_message(user_input) context = self.contexts[self.current_conversation_id] # ==================== 第三步：知识库搜索 ==================== search_results, search_info = self.search_knowledge(user_input, threshold=0.3) # ==================== 第四步：生成答案 ==================== messages_for_llm = [] for msg in context.messages[:-1]: messages_for_llm.append({ "role": msg.role, "content": msg.content }) messages_for_llm.append({ "role": "user", "content": user_input }) # 构建增强的系统提示词 enhanced_prompt = self.system_prompt if search_results: enhanced_prompt += "\n\n【知识库检索结果】\n" for idx, result in enumerate(search_results[:3], 1): enhanced_prompt += f"{idx}. 【{result.item.doc_type}】 相似度: {result.similarity_score:.2%}\n" enhanced_prompt += f" Q: {result.item.question}\n" enhanced_prompt += f" A: {result.item.answer}\n\n" # 尝试使用LLM生成答案 answer = None if self.use_llm and self.ai_client.chat_client: answer = self.ai_client.generate_response( messages=messages_for_llm, system_prompt=enhanced_prompt ) # 如果LLM没有回复或被禁用，使用知识库回复 if not answer: if search_results: # 拼接多个搜索结果 answer = self.answer_formatter.merge_multiple_answers(search_results) else: answer = "" # ==================== 第五步：检查是否需要兜底 ==================== fallback_type = "no_fallback" if not answer or (search_results and search_results[0].similarity_score < 0.3): fallback_response, fallback_type = self.fallback_manager.get_best_fallback_for_search( results=search_results, has_sensitive=False ) if fallback_response: # 组合答案和兜底话术 if answer: final_answer = f"{answer}\n\n---\n\n{fallback_response}" else: final_answer = fallback_response else: final_answer = answer if answer else "抱歉，我暂时无法回答这个问题。" else: final_answer = answer # ==================== 第六步：答案格式化和引用标注 ==================== related_questions = self.answer_formatter.generate_related_questions( user_input, search_results, max_questions=3 ) formatted = self.answer_formatter.format_answer( answer=final_answer, sources=search_results if search_results else None, related_questions=related_questions if related_questions else None ) # ==================== 第七步：存储消息 ==================== assistant_msg = self.add_assistant_message(final_answer) context_summary = context.get_context_summary() return final_answer, { "user_message": user_input, "assistant_message": final_answer, "has_sensitive": False, "is_fallback": fallback_type != "no_fallback", "fallback_type": fallback_type, "sources": formatted["sources"], "related_questions": formatted["related_questions"], "markdown": formatted["markdown"], "search_results_count": len(search_results), "timestamp": datetime.now().isoformat() } # ==================== 意图识别相关 ==================== def get_user_intent(self, message_id: Optional[str] = None) -> Dict: """获取消息的用户意图""" if not self.current_conversation_id: return {"intent": Intent.UNKNOWN.value, "messages": []} context = self.contexts[self.current_conversation_id] user_messages = [m for m in context.messages if m.role == "user"] intents = [] for msg in user_messages: if msg.intent: intents.append({ "message": msg.content, "intent": msg.intent, "timestamp": msg.timestamp }) return { "total_intents": len(intents), "recent_intent": intents[-1]["intent"] if intents else Intent.UNKNOWN.value, "intent_history": intents } # ==================== 检索过滤功能 ==================== def filter_search_results(self, results: List[SearchResult], min_similarity: float = 0.5, doc_types: List[str] = None, max_results: int = 5) -> List[SearchResult]: """过滤搜索结果""" # 按相似度过滤 filtered = [r for r in results if r.similarity_score >= min_similarity] # 按文档类型过滤 if doc_types: filtered = [r for r in filtered if r.item.doc_type in doc_types] # 限制结果数量 filtered = filtered[:max_results] return filtered #
